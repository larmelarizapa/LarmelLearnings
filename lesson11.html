What is a Higher-Order Function?

Description: A higher-order function is a function that either accepts one or more functions as arguments, returns a function as its result, or both.


  function greet(name) {
    return function(message) {
      console.log(message + ", " + name);
    };
  }
  
  let greetLarmel = greet("Larmel");
  greetLarmel("Hello"); // "Hello, Larmel"

Functions as Arguments

Description: A higher-order function can accept another function as an argument and execute it. This allows for greater flexibility and code reuse.


  function processUserData(user, callback) {
    // Simulating processing
    callback(user);
  }
  
  processUserData({name: "Larmel", age: 19}, function(user) {
    console.log("User data processed: " + user.name + " is " + user.age + " years old.");
  });

Functions Returning Functions

Description: A higher-order function can return another function that can later be called with specific arguments. This is a common pattern for creating customized functions.


  function multiplier(factor) {
    return function(number) {
      return number * factor;
    };
  }
  
  let double = multiplier(2);
  console.log(double(5)); // 10
  
  let triple = multiplier(3);
  console.log(triple(5)); // 15

Array Methods as Higher-Order Functions

Description: JavaScript array methods like map(), filter(), and reduce() are examples of higher-order functions because they accept functions as arguments and operate on arrays.


  // Using map to apply a function to each element in an array
  let numbers = [1, 2, 3, 4];
  let doubledNumbers = numbers.map(function(number) {
    return number * 2;
  });
  console.log(doubledNumbers); // [2, 4, 6, 8]
  
  // Using filter to create a new array with elements that satisfy a condition
  let evenNumbers = numbers.filter(function(number) {
    return number % 2 === 0;
  });
  console.log(evenNumbers); // [2, 4]

The map(), filter(), and reduce() Methods

Description: These built-in methods are powerful higher-order functions for processing arrays in JavaScript. They simplify operations like transformation, filtering, and reducing data into a single value.


  // Example of map() - Transforming array values
  let numbers = [1, 2, 3];
  let squares = numbers.map(function(number) {
    return number * number;
  });
  console.log(squares); // [1, 4, 9]
  
  // Example of filter() - Filtering out unwanted values
  let numbers = [1, 2, 3, 4, 5];
  let oddNumbers = numbers.filter(function(number) {
    return number % 2 !== 0;
  });
  console.log(oddNumbers); // [1, 3, 5]
  
  // Example of reduce() - Reducing array to a single value
  let sum = numbers.reduce(function(accumulator, currentValue) {
    return accumulator + currentValue;
  }, 0);
  console.log(sum); // 15

Closures and Higher-Order Functions

Description: A closure is a function that retains access to the variables from its lexical scope, even after the outer function has finished executing. This can be useful when working with higher-order functions.


  function createCounter() {
    let count = 0;
    return function() {
      count++;
      console.log(count);
    };
  }
  
  let counter = createCounter();
  counter(); // 1
  counter(); // 2
  counter(); // 3
