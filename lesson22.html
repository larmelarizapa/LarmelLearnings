What is a Closure?

Description: A closure is a function that retains access to the variables from its lexical scope, even after the outer function has finished executing. This means that the inner function can still access variables from the outer function even when it is executed outside of that function.


  function outer() {
    let message = "Hello, closure!";
    
    function inner() {
      console.log(message);
    }
    
    return inner;
  }
  
  let closureFunction = outer();
  closureFunction(); // "Hello, closure!"

How Closures Work

Description: Closures occur when a function is able to "remember" its lexical scope, which includes variables from the outer function even after the outer function has finished execution. This is particularly useful for data encapsulation and function factories.


  function counter() {
    let count = 0;
    
    return function() {
      count++;
      return count;
    };
  }
  
  let increment = counter();
  console.log(increment()); // 1
  console.log(increment()); // 2
  console.log(increment()); // 3

Closures and Private Variables

Description: Closures are often used to create private variables that cannot be accessed directly from outside the function, providing better data encapsulation.


  function createBankAccount() {
    let balance = 0;
    
    return {
      deposit: function(amount) {
        balance += amount;
        console.log("Deposited: " + amount);
      },
      getBalance: function() {
        return balance;
      }
    };
  }
  
  let account = createBankAccount();
  account.deposit(100);
  console.log(account.getBalance()); // 100
  // balance is private, cannot be accessed directly
  // console.log(account.balance); // undefined

Closures and Functions as Arguments

Description: Functions can accept other functions as arguments, and closures are useful when you want to pass a function with access to the outer function's variables to another function.


  function greet(name) {
    return function(message) {
      console.log(message + ", " + name);
    };
  }
  
  function sendMessage(greetingFunction, message) {
    greetingFunction(message);
  }
  
  let greetLarmel = greet("Larmel");
  sendMessage(greetLarmel, "Hello"); // "Hello, Larmel"

Closures and setTimeout

Description: Closures are commonly used with asynchronous functions like setTimeout to maintain access to the variables in the scope where the function was created, even after a delay.


  function createTimer(seconds) {
    let time = seconds;
    
    setTimeout(function() {
      console.log("Timer finished after " + time + " seconds");
    }, time * 1000);
  }
  
  createTimer(3); // "Timer finished after 3 seconds"

Using Closures with Event Handlers

Description: Closures are often used in event handlers to "remember" certain data, even when the event occurs later on.


  function setupButton(buttonId, message) {
    let button = document.getElementById(buttonId);
    
    button.addEventListener('click', function() {
      alert(message);
    });
  }
  
  setupButton("myButton", "Button clicked!"); // Displays alert when button is clicked
